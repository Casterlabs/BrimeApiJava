```xml
    <repositories>
        <repository>
            <id>jitpack.io</id>

        </repository>
    </repositories>

    <dependencies>
        <dependency>
            <groupId>com.github.casterlabs</groupId>
            <artifactId>BrimeApiJava</artifactId>
            <version>1.0.0</version>
            <scope>compile</scope>
        </dependency>
    </dependencies>
```
